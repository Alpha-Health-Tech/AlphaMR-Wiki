import { getCollection } from 'astro:content';
import { categories } from '../../data/categories';

export const allDocs = await getCollection('docs');

export const featuredArticles = allDocs
  .filter(doc => doc.data.featured === true && doc.slug !== 'index')
  .slice(0, 6);

export const getFirstArticleInCategory = (categoryId) => {
  const article = allDocs.find(doc =>
    doc.slug.startsWith(categoryId + '/') && doc.slug !== 'index'
  );
  return article ? `/AlphaMR-Wiki/${article.slug}/` : `/AlphaMR-Wiki/${categoryId}/`;
};

export const activeCategories = categories.filter(cat =>
  allDocs.some(doc => doc.slug.startsWith(cat.id + '/'))
);

<h2 class="section-title">Browse by Category</h2>
<p class="section-subtitle"><a href="/AlphaMR-Wiki/overview/getting-started/">View all topics â†’</a></p>

<div class="category-grid">
  {activeCategories.map(cat => (
    <a href={getFirstArticleInCategory(cat.id)} class="category-card">
      <div class="category-icon" style={`background: ${cat.iconBg};`}>
        <svg xmlns="http://www.w3.org/2000/svg" width="28" height="28" viewBox="0 0 24 24" fill="none" stroke={cat.iconColor} stroke-width="2" stroke-linecap="round" stroke-linejoin="round" set:html={cat.icon} />
      </div>
      <h3>{cat.label}</h3>
      <p>{cat.description}</p>
    </a>
  ))}
</div>

{featuredArticles.length > 0 && (
  <>
    <h2 class="section-title" style="margin-top: 3rem;">Popular Articles</h2>
    <div class="articles-list">
      {featuredArticles.map(article => (
        <a href={`/AlphaMR-Wiki/${article.slug}/`} class="article-row">
          <svg xmlns="http://www.w3.org/2000/svg" width="20" height="20" viewBox="0 0 24 24" fill="none" stroke="#6D66FE" stroke-width="2" stroke-linecap="round" stroke-linejoin="round">
            <path d="M14.5 2H6a2 2 0 0 0-2 2v16a2 2 0 0 0 2 2h12a2 2 0 0 0 2-2V7.5L14.5 2z"></path>
            <polyline points="14 2 14 8 20 8"></polyline>
          </svg>
          <div>
            <h4>{article.data.title}</h4>
            <p>{article.data.description}</p>
          </div>
        </a>
      ))}
    </div>
  </>
)}
